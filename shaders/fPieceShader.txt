#version 330 core
out vec4 FragColor;

in float textureSelect;
in vec2 texCoord;

uniform sampler2D texture1;
uniform sampler2D texture2;

void main()
{
    if (textureSelect == 1.0)
    {
        FragColor = texture(texture1, texCoord);
    }
    else if (textureSelect == 2.0)
    {
        FragColor = texture(texture2, texCoord);
    }
};
